    PicoCTF Reverse Engineering

    These labs demonstrate basic assembly language understanding.

    Bit-O-Asm-1

    What is in the eax register?

    <+0>:     endbr64 
<+4>:     push   rbp
<+5>:     mov    rbp,rsp
<+8>:     mov    DWORD PTR [rbp-0x4],edi
<+11>:    mov    QWORD PTR [rbp-0x10],rsi
<+15>:    mov    eax,0x30
<+20>:    pop    rbp
<+21>:    ret
    Walkthrough
    1. Look for eax
    2. Notice that it is next to a mov <dst>, <src> instruction
    3. We can see a hex value is moved into eax
    4. Convert the hex value to decimal
    <+0>:     endbr64 
<+4>:     push   rbp
<+5>:     mov    rbp,rsp
<+8>:     mov    DWORD PTR [rbp-0x4],edi
<+11>:    mov    QWORD PTR [rbp-0x10],rsi
<+15>:    mov    eax,0x30     // this is the line of interest
<+20>:    pop    rbp
<+21>:    ret

    Bit-O-Asm-2

    <+0>:     endbr64 
<+4>:     push   rbp
<+5>:     mov    rbp,rsp
<+8>:     mov    DWORD PTR [rbp-0x14],edi
<+11>:    mov    QWORD PTR [rbp-0x20],rsi
<+15>:    mov    DWORD PTR [rbp-0x4],0x9fe1a // set value at memory location
<+22>:    mov    eax,DWORD PTR [rbp-0x4]  // move value at memory location into eax
<+25>:    pop    rbp
<+26>:    ret
    Walkthrough
    1. Look for eax
    2. Notice that it is next to a mov <dst>, <src> instruction
    3. Notice the src is a memory address, but more specifically the value at a memory address rbp-0x4
    4. So we look to where that value is set, which is the prior mov instruction
    5. Grab the hex value and convert to decimal
    <+0>:     endbr64 
<+4>:     push   rbp
<+5>:     mov    rbp,rsp
<+8>:     mov    DWORD PTR [rbp-0x14],edi
<+11>:    mov    QWORD PTR [rbp-0x20],rsi
<+15>:    mov    DWORD PTR [rbp-0x4],0x9fe1a
<+22>:    mov    eax,DWORD PTR [rbp-0x4]      //9fe1a
<+25>:    pop    rbp
<+26>:    ret

    Bit-O-Asm-3

    <+0>:     endbr64 
<+4>:     push   rbp
<+5>:     mov    rbp,rsp
<+8>:     mov    DWORD PTR [rbp-0x14],edi
<+11>:    mov    QWORD PTR [rbp-0x20],rsi
<+15>:    mov    DWORD PTR [rbp-0xc],0x9fe1a
<+22>:    mov    DWORD PTR [rbp-0x8],0x4
<+29>:    mov    eax,DWORD PTR [rbp-0xc]
<+32>:    imul   eax,DWORD PTR [rbp-0x8]
<+36>:    add    eax,0x1f5
<+41>:    mov    DWORD PTR [rbp-0x4],eax
<+44>:    mov    eax,DWORD PTR [rbp-0x4]
<+47>:    pop    rbp
<+48>:    ret
    Walkthrough
    1. Track the value of eax back up through the instructions
    2. notice the instructions are not all mov!
    <+0>:     endbr64 
<+4>:     push   rbp
<+5>:     mov    rbp,rsp
<+8>:     mov    DWORD PTR [rbp-0x14],edi
<+11>:    mov    QWORD PTR [rbp-0x20],rsi
<+15>:    mov    DWORD PTR [rbp-0xc],0x9fe1a // used at <+29>
<+22>:    mov    DWORD PTR [rbp-0x8],0x4    // used at <+32>
<+29>:    mov    eax,DWORD PTR [rbp-0xc] // eax = 0x9fe1a
<+32>:    imul   eax,DWORD PTR [rbp-0x8] // eax = 0x9fe1a * 0x4 = 0x27f868   
<+36>:    add    eax,0x1f5               // eax =  0x27f868 + 0x1f5  = 
<+41>:    mov    DWORD PTR [rbp-0x4],eax // value at eax moved into rbp-0x4
<+44>:    mov    eax,DWORD PTR [rbp-0x4] // value at rbp-0x4 moved into eax
<+47>:    pop    rbp
<+48>:    ret

    Enter the interactive python by running the command python3. You can run some code like:

    a = 0x9fe1a
b = 0x4
c = 0x1f5
ans = (a*b)+c
print(f'picoCTF{{{ans}}}')
